```
import math as A
import numpy as BB
import matplotlib.pyplot as CCC

# –ö–æ–¥, –∫–æ—Ç–æ—Ä—ã–π –¥–æ–ª–∂–µ–Ω –≤—ã–ø–æ–ª–Ω–∏—Ç—å—Å—è
```

–ê–Ω–∞–ª–∏—Ç–∏–∫ –æ—Ç–¥–µ–ª–∞ –ø–æ–ª—É—á–∏–ª –Ω–µ–æ–±—ã—á–Ω—É—é –∑–∞–¥–∞—á—É: –¥–ª—è –∫–æ—Ä–ø–æ—Ä–∞—Ç–∏–≤–∞ –Ω—É–∂–Ω–æ –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–∏—Ü—Ü –∏ –∏—Ö –¥–∏–∞–º–µ—Ç—Ä, —á—Ç–æ–±—ã –≤—Å–µ–º —Ö–≤–∞—Ç–∏–ª–æ, –∏ –Ω–∞–π—Ç–∏ –≤–∞—Ä–∏–∞–Ω—Ç —Å —Å–∞–º–æ–π –≤—ã–≥–æ–¥–Ω–æ–π —Å—Ç–æ–∏–º–æ—Å—Ç—å—é. –í –ø–∏—Ü—Ü–µ—Ä–∏–∏ –¥–æ—Å—Ç—É–ø–Ω—ã —Å–ª–µ–¥—É—é—â–∏–µ –≤–∞—Ä–∏–∞–Ω—Ç—ã –¥–ª—è –∑–∞–∫–∞–∑–∞:  
–î–∏–∞–º–µ—Ç—Ä –ø–∏—Ü—Ü—ã, —Å–º	–°—Ç–æ–∏–º–æ—Å—Ç—å

–û–ø—Ä–µ–¥–µ–ª–∏—Ç–µ –¥–∏–∞–º–µ—Ç—Ä —Å–∞–º–æ–π –≤—ã–≥–æ–¥–Ω–æ–π –ø–∏—Ü—Ü—ã.  
```
## –ø—Ä–µ–∫–æ–¥
# –∏–º–ø–æ—Ä—Ç –±–∏–±–ª–∏–æ—Ç–µ–∫
import math
import numpy as np

# –∑–∞–¥–∞—ë–º –¥–∏–∞–º–µ—Ç—Ä—ã
diams = [24, 30, 35, 40]
# –∑–∞–¥–∞—ë–º —Å—Ç–æ–∏–º–æ—Å—Ç–∏
prices = [550, 690, 800, 1100]
# —Å–æ–∑–¥–∞—ë–º —Å–ø–∏—Å–æ–∫, –≤ –∫–æ—Ç–æ—Ä—ã–π –ø–æ–ª–æ–∂–∏–º –ø–æ–ª—É—á–µ–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è
result_list = []

for i in range(0,4): # —Ü–∏–∫–ª –ø–æ –≤–æ–∑–º–æ–∂–Ω—ã–º –≤–∞—Ä–∏–∞–Ω—Ç–∞–º
    d = diams[i] # –¥–∏–∞–º–µ—Ç—Ä, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π –≤–∞—Ä–∏–∞–Ω—Ç—É
    price = prices[i] # —Ü–µ–Ω–∞, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∞—è –≤–∞—Ä–∏–∞–Ω—Ç—É

    ## –º–µ—Å—Ç–æ –¥–ª—è –≤–∞—à–µ–≥–æ –∫–æ–¥–∞
    # –≤—ã—á–∏—Å–ª—è–µ–º —Ä–∞–¥–∏—É—Å
    radius = d / 2

    # –≤—ã—á–∏—Å–ª—è–µ–º –ø–ª–æ—â–∞–¥—å 
    s = math.pi * math.pow(radius ,2)

    # –≤—ã—á–∏—Å–ª—è–µ–º —Å–æ–æ—Ç–Ω–æ—à–µ–Ω–∏–µ —Ü–µ–Ω—ã –∫ –ø–ª–æ—â–∞–¥–∏
    result = price / s

    ## –ø—Ä–µ–∫–æ–¥
    # –∑–∞–ø–∏—Å—ã–≤–∞–µ–º –ø–æ–ª—É—á–µ–Ω–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤ —Å–ø–∏—Å–æ–∫ –¥–ª—è –¥–∞–ª—å–Ω–µ–π—à–µ–≥–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è
    result_list.append(result)

## –º–µ—Å—Ç–æ –¥–ª—è –≤–∞—à–µ–≥–æ –∫–æ–¥–∞
# –Ω–∞—Ö–æ–¥–∏–º –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ result_list
optimal_result = min(result_list)

## –ø—Ä–µ–∫–æ–¥
# –¥–∏–∞–º–µ—Ç—Ä, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π –Ω–æ–º–µ—Ä—É –≤–∞—Ä–∏–∞–Ω—Ç–∞
# —Å –º–∏–Ω–∏–º–∞–ª—å–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º result_list

print('–ù–∞–∏–±–æ–ª–µ–µ –≤—ã–≥–æ–¥–Ω—ã–π –≤–∞—Ä–∏–∞–Ω—Ç - –ø–∏—Ü—Ü–∞ —Å –¥–∏–∞–º–µ—Ç—Ä–æ–º', optimal_diam, '—Å–º')

```

```
import math

# –∑–∞–¥–∞—ë–º –≤—ã–±—Ä–∞–Ω–Ω—ã–π –¥–∏–∞–º–µ—Ç—Ä

# —Ü–µ–Ω–∞, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∞—è –≤—ã–±—Ä–∞–Ω–Ω–æ–º—É –¥–∏–∞–º–µ—Ç—Ä—É

# —Å—á–∏—Ç–∞–µ–º —Å–∫–æ–ª—å–∫–æ –≤—Å–µ–≥–æ –ø–∏—Ü—Ü—ã –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ, —á—Ç–æ–±—ã –∫–∞–∂–¥–æ–º—É –∏–∑ —Å–æ—Ç—Ä—É–¥–Ω–∏–∫–æ–≤

# –≤—ã—á–∏—Å–ª—è–µ–º —Ä–∞–¥–∏—É—Å

# –ø–ª–æ—â–∞–¥—å –æ–¥–Ω–æ–π –ø–∏—Ü—Ü—ã —Å –≤—ã–±—Ä–∞–Ω–Ω—ã–º –¥–∏–∞–º–µ—Ç—Ä–æ–º

# –æ–ø—Ä–µ–¥–µ–ª–∏–º –ø–æ–ª–Ω—É—é —Å—Ç–æ–∏–º–æ—Å—Ç—å –∑–∞–∫–∞–∑–∞
```
